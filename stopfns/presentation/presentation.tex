% --- Bayesian view: the PPD -----------------------------------------------
\begin{frame}{Posterior Predictive Distribution}
\begin{columns}[T,onlytextwidth]
\begin{column}{0.54\textwidth}
  \vspace{0.6em}
  \gd{PPD} averages every model,
  \[ {\color{good}\pi(y\mid\bx,\Dn)} = \int p(y\mid\bx)\,d\pp{\Pi(p\mid\Dn)} \]
  \vspace{2em}

  weighted by its \pp{posterior}

  $\pp{\Pi(p\mid\Dn)}$: how plausible model $p$ is, given $\Dn$
\end{column}
\begin{column}{0.44\textwidth}
\centering
\begin{tikzpicture}
\begin{axis}[
  width=6.2cm, height=5.4cm,
  axis lines=left, clip=false,
  xmin=0, xmax=6, ymin=0, ymax=3.2,
  xtick=\empty, ytick=\empty,
  xlabel={$\bx$}, ylabel={$y$},
  x label style={at={(axis description cs:1,0)}, anchor=north east},
  y label style={at={(axis description cs:0,1)}, anchor=south east, rotate=-90},
]
  % low-posterior models (poor fit): faint
  \addplot[muted, opacity=0.35, line width=0.5pt, domain=0.3:5.7, samples=70]{1.0+0.28*sin(60*x)};
  \addplot[muted, opacity=0.35, line width=0.5pt, domain=0.3:5.7, samples=70]{2.2+0.28*sin(60*x+90)};
  % medium-posterior models
  \addplot[muted, opacity=0.7, line width=0.7pt, domain=0.3:5.7, samples=70]{1.32+0.26*sin(60*x+40)};
  \addplot[muted, opacity=0.7, line width=0.7pt, domain=0.3:5.7, samples=70]{1.95+0.26*sin(60*x+120)};

% --- posterior consistency (Theorem 3.1) ----------------------------------
\begin{frame}{Posterior consistency}
\begin{columns}[T,onlytextwidth]
\begin{column}{0.52\textwidth}
  \begin{block}{Theorem 3.1}
    Under \mut{(A1)--(A2)}, as $n\to\infty$,
    \[ {\color{good}\pi(y\mid\bx,\Dn)} \longrightarrow p^*(y\mid\bx) \quad\text{almost surely,} \]
    with $p^*$ the \gd{KL-optimal} model in $\mathcal{P}$.
  \end{block}
  \vspace{0.7em}

  \footnotesize
  \textbf{(A1)} unique $p^*\in\mathcal{P}$ minimizing $\mathrm{KL}(p^*\,\|\,p_0)<\infty$\\
  \mut{a reachable best target}\\[0.5em]
  \textbf{(A2)} $\mathcal{P}$ covered by small balls of summable prior mass\\
  \mut{prior covers its support well}
\end{column}
\begin{column}{0.46\textwidth}
\centering
\vspace{1.6em}

\begin{tikzpicture}[scale=0.82]
  % support of the prior
  \draw[muted, fill=muted!12, line width=0.8pt]
    plot[smooth cycle, tension=0.8] coordinates
    {(-2.4,0.1) (-1.7,1.2) (0,1.5) (1.5,1.1) (2.1,0.0) (1.3,-1.2) (-0.3,-1.4) (-1.9,-0.9)};
  \node[muted, font=\scriptsize] at (-1.55,1.0) {support $\mathcal{P}$};
  % candidate models in the support
  \foreach \p in {(-1.4,0.3),(-0.7,-0.6),(-0.2,0.7),(0.5,0.1),(0.2,-0.85),(0.9,0.8)}
    \draw[muted, fill=white, line width=0.6pt] \p circle (2pt);
  % the truth, outside the support
  \node[star, star points=5, star point ratio=2.3, fill=ink, draw=ink, inner sep=1.3pt] (p0) at (3.7,0.15) {};
  \node[ink, anchor=south, font=\small] at (3.7,0.45) {$p_0$ (truth)};
  % KL-closest point in the support
  \draw[muted, dashed, line width=0.7pt] (1.25,0.05) -- (p0);
  \node[muted, font=\scriptsize, anchor=south] at (2.45,0.18) {closest in KL};
  \fill[accent] (1.25,0.05) circle (2.6pt);
  \node[accent, anchor=north, font=\small] at (1.25,-0.12) {$p^*$};
\end{tikzpicture}
\end{column}
\end{columns}
\vspace{0.4em}
\gd{takeaway:} more data $\Rightarrow$ PPD $\to$ best model the prior allows
\mut{(even if $p_0\notin\mathcal{P}$)}
\end{frame}
