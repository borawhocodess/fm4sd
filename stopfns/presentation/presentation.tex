% --- G6: diminishing sensitivity (intuition + result) ----------------------
\begin{frame}[t]{Diminishing sensitivity}

  \onslide<1->{\centering Change \emph{one} of the $n$ data points --- how much does the prediction move?\\[2pt]
    {\footnotesize\mut{attention weights sum to $1$, so each sample gets weight $\approx 1/n$ --- more rows, the weight spreads out:}}\par}
  \vspace{0.5em}

  \begin{columns}[T]
    \column{0.5\textwidth}
      \onslide<1->{\centering \textbf{few samples}\\[4pt]
      \begin{tikzpicture}[scale=0.55]
        \draw[muted,line width=0.4pt](-0.1,0)--(2.7,0);
        \foreach \i in {0,1,2,3} \fill[accent] (\i*0.62,0) rectangle (\i*0.62+0.42,1.8);
      \end{tikzpicture}\\[2pt]
      {\footnotesize\mut{flip one $\Rightarrow$ big swing $(\approx\tfrac14)$}}}
    \column{0.5\textwidth}
      \onslide<1->{\centering \textbf{many samples}\\[4pt]
      \begin{tikzpicture}[scale=0.55]
        \draw[muted,line width=0.4pt](-0.1,0)--(3.1,0);
        \foreach \i in {0,...,13} \fill[accent] (\i*0.21,0) rectangle (\i*0.21+0.13,0.5);
      \end{tikzpicture}\\[2pt]
      {\footnotesize\mut{flip one $\Rightarrow$ tiny swing $(\approx\tfrac1n)$}}}
  \end{columns}
  \vspace{0.5em}

  \onslide<2->{\centering
    Swap one sample $\mathcal D_n\to\textcolor{accent}{\mathcal D_n'}$ --- the prediction moves by \emph{at most}:\\[5pt]
    $\bigl|\,q_{\hat\theta}(\mathcal D_n)-q_{\hat\theta}(\textcolor{accent}{\mathcal D_n'})\,\bigr|\;\le\;\textcolor{good!55!black}{L}\;\textcolor{biasc}{n^{-\alpha}}$\par}
  \vspace{0.5em}

  \onslide<3->{\centering {\footnotesize
    {\color{accent}$\mathcal D_n'$}: $\mathcal D_n$ with \textbf{one sample swapped}\quad
    {\color{good!55!black}$L$}: a fixed constant\quad
    {\color{biasc}$n^{-\alpha}$}: \textbf{shrinks as $n$ grows} \,($\alpha$ $=$ how fast)\par}
    \smallskip
    {\footnotesize\mut{$\Rightarrow$ more rows $\Rightarrow$ each single sample's influence \hl{diminishes}.}}\par}
\end{frame}

% --- G9: transformers aren't local -----------------------------------------
\begin{frame}[t]{Transformers aren't local}

  \onslide<1->{\centering Attention is a \hl{soft} weighting: it favours samples similar to $x$, but never drops the far ones to zero.\par}
  \vspace{0.6em}

  \onslide<1->{%
  \begin{columns}[T]
    \column{0.5\textwidth}\centering \textbf{\footnotesize attention (soft)}\\[4pt]
      \begin{tikzpicture}[>=Latex, scale=0.95]
        \draw[->](0,0)--(3.0,0) node[right,font=\tiny]{distance};
        \draw[->](0,0)--(0,2.0) node[above,font=\tiny]{weight};
        \draw[accent, line width=1pt] (0.1,1.75) .. controls (0.9,0.7) and (1.8,0.45) .. (2.8,0.42);
        \node[ink, font=\tiny, align=center] at (2.0,1.1){never\\ hits 0};
      \end{tikzpicture}
    \column{0.5\textwidth}\centering \textbf{\footnotesize a local window (hard)}\\[4pt]
      \begin{tikzpicture}[>=Latex, scale=0.95]
        \draw[->](0,0)--(3.0,0) node[right,font=\tiny]{distance};
        \draw[->](0,0)--(0,2.0) node[above,font=\tiny]{weight};
        \draw[good!60!black, line width=1pt] (0.1,1.35)--(1.15,1.35)--(1.15,0.0)--(2.8,0.0);

  \onslide<2->{\centering Far points \hl{always leak in} $\Rightarrow$ you \emph{can't} flip them freely $\Rightarrow$ \hl{not local}.\par}

\end{frame}

\begin{frame}[t]{Localized PFNs}

  \onslide<1->{\centering A simple add-on to \emph{any} frozen PFN --- localize at prediction time.\par}
  \vspace{0.7em}

  \onslide<1->{\centering
  \begin{tikzpicture}[>=Latex, font=\footnotesize, scale=0.92]

    % query
    \fill[ink] (1.4,1.2) circle(0.06);
    \node[ink, font=\scriptsize, anchor=east] at (1.28,1.2){$x$};
    % near samples (red) = the k_n nearest, inside the ring
    \foreach \p in {(1.18,1.5),(1.72,1.4),(1.66,0.96),(1.12,0.92)} \fill[accent] \p circle(0.05);
    \node[accent, font=\tiny, anchor=east] at (1.62,1.42){$X_i$};
    % far samples (grey) = ignored, outside the ring
    \foreach \p in {(0.42,0.45),(2.62,0.55),(0.5,2.0),(2.55,1.95)} \fill[muted] \p circle(0.05);
    \node[muted, font=\tiny, anchor=south west] at (2.66,0.6){$X_j$};
    \node[muted, font=\tiny, anchor=north] at (1.4,-0.12){keep $k_n$ nearest to $x$};
    \draw[->,line width=0.8pt] (3.2,1.2)--(3.95,1.2);
    \node[draw=ink, fill=ink!5, rounded corners, minimum width=14mm, minimum height=10mm] (q) at (4.75,1.2) {$q_{\hat\theta}$};
    \draw[->,line width=0.8pt] (5.55,1.2)--(6.3,1.2);
    \node at (7.05,1.2) {$\hat q(\cdot\mid x)$};
  \end{tikzpicture}}
  \vspace{0.4em}

  \onslide<1->{\centering {\scriptsize {\color{ink}$\bullet$}~$x$ (query)\quad
    {\color{accent}$\bullet$}~near samples $X_i$ (kept)\quad {\color{muted}$\bullet$}~far samples $X_j$ (dropped)}\par}
  \vspace{0.4em}

  \onslide<2->{\centering \textbf{Why it works:} near $x$ the truth is $\approx$ \hl{flat} $\Rightarrow$ easy to fit $\Rightarrow$ \gd{bias drops}. \;\mut{\footnotesize (cost: fewer points $\Rightarrow$ slightly more variance.)}\par}
  \vspace{0.2em}

  \onslide<3->{\centering Does it actually break the bias floor? \;\mut{\footnotesize We'll put it to the test at the end \cit{Fig.\ 1}.}\par}
\end{frame}

\begin{frame}[t]{Specific PFNs}

  \onslide<1->{\centering Two more concrete predictors, each with a different bias/variance behaviour:\par}
  \vspace{2.0em}

  {\large
  \begin{enumerate}\setlength\itemsep{1.2em}
    \item \textbf{Window smoother}
    \item \textbf{Classification trees}

% --- 1. Window smoother (formula explained notation-by-notation) -----------
\begin{frame}[t]{Window smoother}

  \onslide<1->{\centering
  \[
    q_\theta(y\mid x,\mathcal D_n)\;=\;
    \frac{\sum_{i}\, \alt<3->{\textcolor{accent}{\mathbf{1}(Y_i=y)}}{\textcolor{black!30}{\mathbf{1}(Y_i=y)}}\;
                       \alt<2->{\textcolor{good!55!black}{\mathbf{1}(\|X_i-x\|<\theta)}}{\textcolor{black!30}{\mathbf{1}(\|X_i-x\|<\theta)}}}
         {\sum_{i}\, \alt<2->{\textcolor{good!55!black}{\mathbf{1}(\|X_i-x\|<\theta)}}{\textcolor{black!30}{\mathbf{1}(\|X_i-x\|<\theta)}}}
  \]\par}
  \vspace{0.4em}

  \begin{columns}[T]
    \column{0.44\textwidth}
      \onslide<1->{\centering
      \begin{tikzpicture}[>=Latex]
        \draw[good, dashed, line width=0.9pt] (0,0) circle(1.2);
        \draw[good!55!black, line width=0.5pt] (0,0) -- (1.2,0) node[midway, below, font=\scriptsize]{$\theta$};
        \fill[ink] (0,0) circle(0.06);
        \node[ink, font=\scriptsize, below=0pt] at (0,-0.04){$x$};
        \fill[accent] (0.45,0.5) circle(0.075);
        \node[accent, font=\scriptsize, above right=-3pt] at (0.45,0.5){$X_i$};
        \foreach \p in {(-0.55,0.35),(0.25,-0.6)} \fill[accent] \p circle(0.075);
        \fill[biasc] (-0.45,-0.45) circle(0.075);
        \foreach \p in {(1.75,0.7),(-1.8,0.5),(1.6,-0.9),(-1.5,-1.05),(0.1,1.75)} \fill[muted] \p circle(0.075);
      \end{tikzpicture}\\[4pt]
      {\scriptsize {\color{accent}$\bullet$}~label $y$\quad {\color{biasc}$\bullet$}~other\quad {\color{muted}$\bullet$}~outside}}
    \column{0.56\textwidth}
      {\footnotesize
      \onslide<1->{dots $=$ training points $X_i$ (with label $Y_i$);\; $x$ $=$ the query.\par}\medskip
      \onslide<2->{{\color{good!55!black}$\|X_i-x\|<\theta$}: $X_i$ is \textbf{inside the window}.\par}\medskip
      \onslide<3->{{\color{accent}$\mathbf{1}(Y_i=y)$}: its label is $y$ (red).\par}\medskip
      \onslide<4->{$q_\theta=\dfrac{\#\{\text{inside \& }y\}}{\#\{\text{inside}\}}=\dfrac{3}{4}$.\par}}
  \end{columns}
\end{frame}

% --- 1b. Window smoother: shrink the window -> bias -> 0 -------------------
\begin{frame}[t]{Window smoother --- shrink the window, kill the bias}

  \onslide<1->{\centering Now let the radius \emph{shrink with $n$}: replace $\theta$ by $\textcolor{accent}{a_n}\,\theta$.
  \[
    q_\theta(y\mid x,\mathcal D_n)=
    \frac{\sum_i \mathbf{1}(Y_i=y)\,\mathbf{1}(\|X_i-x\|<\textcolor{accent}{a_n}\theta)}
         {\sum_i \mathbf{1}(\|X_i-x\|<\textcolor{accent}{a_n}\theta)}.
  \]\par}
  \vspace{1.2em}

  \onslide<2->{\centering Optimal schedule \;\hl{$a_n=n^{-1/(4+d)}$}\; \mut{\footnotesize($d=$ \#features)}: \;\gd{bias $\to 0$ as data grows}.\par}
\end{frame}

% --- 2a. Classification trees: what they are + how they predict -----------
\begin{frame}[t]{Classification trees}

  \onslide<1->{\centering A tree \hl{cuts feature space into bins}; to predict at $x$, report the \hl{label-mix} of $x$'s bin.\par}
  \vspace{0.7em}

  % --- 1-D number-line bins (the nice visualization) ---
  \onslide<1->{\centering
  \begin{tikzpicture}[>=Latex, scale=1.0]
    \fill[good!12] (3.0,0) rectangle (5.4,0.95);
    \draw[->] (0,0) -- (8.0,0) node[right, font=\scriptsize]{feature axis};
    \foreach \tx/\lab in {3.0/$\theta_1$, 5.4/$\theta_2$}{
      \draw[muted, line width=0.6pt] (\tx,-0.08)--(\tx,0.95);
      \node[muted, font=\scriptsize, below] at (\tx,-0.1){\lab};}
    \node[muted, font=\scriptsize] at (1.5,1.12){bin 1};
    \node[good!55!black, font=\scriptsize] at (4.2,1.12){bin 2};
    \node[muted, font=\scriptsize] at (6.7,1.12){bin 3};
    \foreach \p in {(0.8,0.4),(1.9,0.55),(2.4,0.3)} \fill[muted] \p circle(0.07);
    \foreach \p in {(3.4,0.5),(4.1,0.35),(4.8,0.55),(5.1,0.3)} \fill[accent] \p circle(0.07);
    \foreach \p in {(6.0,0.45),(7.2,0.5)} \fill[muted] \p circle(0.07);
    \fill[ink] (4.3,0) circle(0.09); \node[ink, font=\scriptsize, below] at (4.3,-0.06){$x$};
  \end{tikzpicture}\\[2pt]
  {\footnotesize\mut{$x$ lands in \gd{bin 2} $\Rightarrow$ use only the points there.}}}
  \vspace{0.6em}

  % --- the prediction formula, decoded term by term ---
  \onslide<2->{\centering
  \[
    q_\theta(y\mid x,\mathcal D_n)=\sum_{j}\;
      \alt<2->{\textcolor{good!55!black}{\mathbf 1(x\in \mathrm{bin}_j)}}{\textcolor{black!30}{\mathbf 1(x\in \mathrm{bin}_j)}}\;
      \frac{\sum_i \alt<4->{\textcolor{accent}{\mathbf 1(Y_i=y)}}{\textcolor{black!30}{\mathbf 1(Y_i=y)}}\,
             \alt<3->{\textcolor{biasc}{\mathbf 1(X_i\in \mathrm{bin}_j)}}{\textcolor{black!30}{\mathbf 1(X_i\in \mathrm{bin}_j)}}}
           {\sum_i \alt<3->{\textcolor{biasc}{\mathbf 1(X_i\in \mathrm{bin}_j)}}{\textcolor{black!30}{\mathbf 1(X_i\in \mathrm{bin}_j)}}}
  \]\par}

  \onslide<2->{\centering\footnotesize
    {\color{good!55!black}which bin holds $x$}%
    \onslide<3->{\;\;\; {\color{biasc}the points in it}}%
    \onslide<4->{\;\;\; {\color{accent}\dots\ with label $y$} $\;\Rightarrow\;$ the bin's \textbf{label fraction}}\par}
\end{frame}

% --- 2b. Classification trees: bias is stuck -> vote among many trees ------
\begin{frame}[t]{Classification trees: the bias is stuck}

  \onslide<1->{\centering
  \[ \text{bias}=\underbrace{\Pr_0(Y=y\mid X\in\mathrm{bin})}_{\text{truth averaged over the bin}}
      -\underbrace{p_0(y\mid x)}_{\text{truth at }x} \]\par}

  \onslide<1->{\centering Bins are \hl{fixed} $\Rightarrow$ each bin blurs the same patch $\Rightarrow$ \hl{bias can't shrink} \mut{\footnotesize(more data $=$ tighter, not truer)}.\par}
  \vspace{0.5em}

  \onslide<2->{\centering \gd{Fix:} average \hl{many trees} with \emph{different} splits:\; $q=\sum_k w_k\,q_k$.\par}
  \vspace{0.2em}

  \begin{columns}[T]
    \column{0.5\textwidth}\centering
      \onslide<2->{\textbf{\footnotesize many different trees}\\[6pt]
      \begin{tikzpicture}[>=Latex, scale=0.72, font=\tiny]
        \draw[muted] (0,1.4)--(2.6,1.4); \node[muted, left] at (-0.1,1.4){$T_1$};
        \draw[muted] (1.2,1.32)--(1.2,1.48);
        \draw[muted] (0,0.7)--(2.6,0.7); \node[muted, left] at (-0.1,0.7){$T_2$};
        \foreach \s in {0.7,1.4,2.0} \draw[muted] (\s,0.62)--(\s,0.78);
        \draw[muted] (0,0.0)--(2.6,0.0); \node[muted, left] at (-0.1,0.0){$T_3$};
        \foreach \s in {0.9,1.8} \draw[muted] (\s,-0.08)--(\s,0.08);
      \end{tikzpicture}}
    \column{0.5\textwidth}\centering
      \onslide<3->{\textbf{\footnotesize weight $w_k=$ how well tree $k$ fits $\mathcal D_n$}\\[6pt]
      \begin{tikzpicture}[scale=0.72, font=\tiny]
        \fill[muted] (0,0) rectangle (0.45,0.18);
        \fill[good]  (0.75,0) rectangle (1.2,1.4);
        \fill[muted] (1.5,0) rectangle (1.95,0.13);
        \draw[->](-0.12,0)--(2.2,0);
        \node[muted, below] at (0.225,-0.02){$T_1$};
        \node[good!55!black, below] at (0.975,-0.02){$T_2$};
        \node[muted, below] at (1.725,-0.02){$T_3$};
      \end{tikzpicture}\\[1pt]
      {\scriptsize\mut{predicts the real labels better $\Rightarrow$ bigger weight}}}
  \end{columns}
  \vspace{0.3em}

  \onslide<4->{\centering {\gd{$n$ grows $\Rightarrow$ best-fitting tree wins $\Rightarrow$ bias drops.}}\par}
\end{frame}
